#include<LPC214X.h>
void  delay(){
	unsigned int i=0;
	while(i<200000){
		i++;
	}
}
int main(){
	//set the direction as output
	IODIR0 |=(0x01<<16);
	IODIR1 &=~(0x01<<16);
	while(1){
	
	if((IO1PIN & (1<<16))==0){
	//turn on led
		IOSET0 |=(0x01<<16);
		}
		else {
		//Turn off LED
		IOCLR0 |=(0x01<<16);
		}
	delay();
	}
}
